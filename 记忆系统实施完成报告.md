### 2. 核心算法实现 ✅

#### 2.1 衰减算法
```javascript
function applyDecay(memory) {
  if (memory.isPermanent) return memory.weight;
  const daysPassed = getDaysSince(memory.lastAccessed);
  const newWeight = memory.weight * Math.exp(-memory.decayRate * daysPassed);
  return Math.max(0, newWeight);
}
```

**衰减速率**：
- Critical: 0（不衰减）
- High: 0.01（100天后衰减到 37%）
- Medium: 0.05（20天后衰减到 37%）
- Low: 0.15（7天后衰减到 37%）

#### 2.2 强化算法
```javascript
function reinforceMemory(memory, reason = 'access') {
  const bonuses = {
    access: 10,       // 普通访问
    repeat: 15,       // 重复提及
    emotion: 20,      // 情感共鸣
    correction: 25,   // 用户纠正
  };
  memory.weight = Math.min(100, memory.weight + bonuses[reason]);
  memory.accessCount++;
  // 访问3次以上升级为重要记忆
  if (memory.accessCount >= 3 && memory.importance === 'medium') {
    memory.importance = 'high';
  }
}
```

#### 2.3 记错算法
```javascript
function shouldMakeMistake(memory) {
  if (memory.isPermanent) return false;
  const fuzzyLevel = 100 - memory.weight;
  let probability = 0;
  if (fuzzyLevel >= 70) probability = 0.4;      // 40%
  else if (fuzzyLevel >= 50) probability = 0.2; // 20%
  else if (fuzzyLevel >= 30) probability = 0.1; // 10%
  else probability = 0.02;                      // 2%
  if (memory.correctedCount > 0) probability *= 0.5;
  return Math.random() < probability;
}
```

#### 2.4 重要性判断
```javascript
function detectImportance(text, emotion) {
  const criticalKeywords = [
    '第一次', '初次', '我喜欢你', '我爱你', '在一起',
    '生日', '纪念日', '保证', '发誓', '一定会',
  ];
  const highKeywords = [
    '喜欢', '讨厌', '最爱', '考试', '面试', '旅行', '约好',
  ];
  if (criticalKeywords.some(k => text.includes(k))) return 'critical';
  if (highKeywords.some(k => text.includes(k))) return 'high';
  return 'medium';
}
```

#### 3.3 消息发送时
```javascript
// 检测用户纠正
if (MemorySystem.detectCorrection(text)) {
  const lastMemory = keyMemories[keyMemories.length - 1];
  const correctedMemory = MemorySystem.handleCorrection(lastMemory, text);
  keyMemories[keyMemories.length - 1] = correctedMemory;
  localStorage.setItem('ai-companion-key-memory', JSON.stringify(keyMemories));
}

// 检测重复提及并强化
for (let i = 0; i < keyMemories.length; i++) {
  const memory = keyMemories[i];
  const keywords = memory.content.split(/[，。、\s]+/).filter(k => k.length > 1);
  if (keywords.some(k => text.includes(k))) {
    keyMemories[i] = MemorySystem.reinforceMemory(memory, 'repeat');
    break;
  }
}
```
